-- ------------------------------------------- create table
CREATE TABLE users(
    userId INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
    username TEXT NOT NULL UNIQUE,
    userpw TEXT,
    userNote TEXT,
    userPfp TEXT
);
CREATE TABLE sems (
    semId INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
    semName TEXT NOT NULL,
    semNote TEXT,
    userId INTEGER NOT NULL,
    FOREIGN KEY (userId) REFERENCES users(userId)
);
CREATE TABLE classes (
    classId INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
    className TEXT NOT NULL,
    classCode TEXT,
    classCredit REAL,
    classGradeA REAL,
    classGradeB REAL,
    classGradeC REAL,
    classActualGrade TEXT,
    classNote TEXT,
    semId INTEGER NOT NULL,
    FOREIGN KEY (semId) REFERENCES sems(semId)
);
CREATE TABLE parts (
    partId INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
    partName TEXT NOT NULL,
    partPercent REAL NOT NULL,
    partNote TEXT,
    classId INTEGER NOT NULL,
    FOREIGN KEY (classId) REFERENCES classes(classId)
);
CREATE TABLE grades (
    gradeId INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
    gradeGot REAL,
    gradeOutOf REAL,
    gradeGotEst REAL,
    gradeOutOfEst REAL,
    gradeNote TEXT,
    gradeDD INTEGER,
    gradeMM INTEGER,
    gradeYY INTEGER,
    partId INTEGER NOT NULL,
    FOREIGN KEY (partId) REFERENCES parts(partId)
);
